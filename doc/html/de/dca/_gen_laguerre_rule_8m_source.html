291 function [ aj, bj, zemu ] = class_matrix ( kind, m, alpha, beta )

293 %*****************************************************************************80
294 %
295 %% CLASS_MATRIX computes the Jacobi matrix for a quadrature rule.
296 %
297 % Discussion:
298 %
299 % This routine computes the diagonal AJ and subdiagonal BJ
300 % elements of the order M tridiagonal symmetric Jacobi matrix
301 % associated with the polynomials orthogonal with respect to
302 % the weight function specified by KIND.
303 %
304 % For weight functions 1-7, M elements are defined in BJ even
305 % though only M-1 are needed. For weight function 8, BJ(M) is
306 % set to zero.
307 %
308 % The zero-th moment of the weight function is returned in ZEMU.
309 %
310 % Licensing:
311 %
312 % This code is distributed under the GNU LGPL license.
313 %
314 % Modified:
315 %
316 % 04 January 2010
317 %
318 % Author:
319 %
320 % Original FORTRAN77 version by Sylvan Elhay, Jaroslav Kautsky.
321 % MATLAB version by John Burkardt.
322 %
323 % Reference:
324 %
325 % Sylvan Elhay, Jaroslav Kautsky,
326 % Algorithm 655: IQPACK, FORTRAN Subroutines for the Weights of
327 % Interpolatory Quadrature,
328 % ACM Transactions on Mathematical Software,
329 % Volume 13, Number 4, December 1987, pages 399-415.
330 %
331 % Parameters:
332 %
333 % Input, integer KIND, the rule.
334 % 1, Legendre, (a,b) 1.0
335 % 2, Chebyshev, (a,b) ((b-x)*(x-a))^(-0.5)
336 % 3, Gegenbauer, (a,b) ((b-x)*(x-a))^alpha
337 % 4, Jacobi, (a,b) (b-x)^alpha*(x-a)^beta
338 % 5, Generalized Laguerre, (a,inf) (x-a)^alpha*exp(-b*(x-a))
339 % 6, Generalized Hermite, (-inf,inf) |x-a|^alpha*exp(-b*(x-a)^2)
340 % 7, Exponential, (a,b) |x-(a+b)/2.0|^alpha
341 % 8, Rational, (a,inf) (x-a)^alpha*(x+b)^beta
342 %
343 % Input, integer M, the order of the Jacobi matrix.
344 %
345 % Input, real ALPHA, the value of Alpha, if needed.
346 %
347 % Input, real BETA, the value of Beta, if needed.
348 %
349 % Output, real AJ(M), BJ(M), the diagonal and subdiagonal
350 % of the Jacobi matrix.
351 %
352 % Output, real ZEMU, the zero-th moment.
353 %
354  temp = eps;
355 
356  parchk ( kind, 2 * m - 1, alpha, beta );
357 
358  temp2 = 0.5;
359 
360  if ( 500.0 * temp < abs ( ( gamma ( temp2 ) )^2 - pi ) )
361  fprintf ( 1, '\n' );
362  fprintf ( 1, 'CLASS - Fatal error!\n' );
363  fprintf ( 1, ' Gamma function does not match machine parameters.\n' );
364  error ( 'CLASS - Fatal error!' );
365  end
366 
367  bj = zeros(m,1);
368  aj = zeros(m,1);
369 
370  if ( kind == 1 )
371 
372  ab = 0.0;
373 
374  zemu = 2.0 / ( ab + 1.0 );
375 
376  aj(1:m) = 0.0;
377 
378  for i = 1 : m
379  abi = i + ab * mod ( i, 2 );
380  abj = 2 * i + ab;
381  bj(i) = abi * abi / ( abj * abj - 1.0 );
382  end
383  bj(1:m) = sqrt ( bj(1:m) );
384 
385  elseif ( kind == 2 )
386 
387  zemu = pi;
388 
389  aj(1:m) = 0.0;
390 
391  bj(1) = sqrt ( 0.5 );
392  bj(2:m) = 0.5;
393 
394  elseif ( kind == 3 )
395 
396  ab = alpha * 2.0;
397  zemu = 2.0^( ab + 1.0 ) * gamma ( alpha + 1.0 )^2 ...
398  / gamma ( ab + 2.0 );
399 
400  aj(1:m) = 0.0;
401  bj(1) = 1.0 / ( 2.0 * alpha + 3.0 );
402  for i = 2 : m
403  bj(i) = i * ( i + ab ) / ( 4.0 * ( i + alpha )^2 - 1.0 );
404  end
405  bj(1:m) = sqrt ( bj(1:m) );
406 
407  elseif ( kind == 4 )
408 
409  ab = alpha + beta;
410  abi = 2.0 + ab;
411  zemu = 2.0^( ab + 1.0 ) * gamma ( alpha + 1.0 ) ...
412  * gamma ( beta + 1.0 ) / gamma ( abi );
413  aj(1) = ( beta - alpha ) / abi;
414  bj(1) = 4.0 * ( 1.0 + alpha ) * ( 1.0 + beta ) ...
415  / ( ( abi + 1.0 ) * abi * abi );
416  a2b2 = beta * beta - alpha * alpha;
417 
418  for i = 2 : m
419  abi = 2.0 * i + ab;
420  aj(i) = a2b2 / ( ( abi - 2.0 ) * abi );
421  abi = abi^2;
422  bj(i) = 4.0 * i * ( i + alpha ) * ( i + beta ) * ( i + ab ) ...
423  / ( ( abi - 1.0 ) * abi );
424  end
425  bj(1:m) = sqrt ( bj(1:m) );
426 
427  elseif ( kind == 5 )
428 
429  zemu = gamma ( alpha + 1.0 );
430 
431  for i = 1 : m
432  aj(i) = 2.0 * i - 1.0 + alpha;
433  bj(i) = i * ( i + alpha );
434  end
435  bj(1:m) = sqrt ( bj(1:m) );
436 
437  elseif ( kind == 6 )
438 
439  zemu = gamma ( ( alpha + 1.0 ) / 2.0 );
440 
441  aj(1:m) = 0.0;
442 
443  for i = 1 : m
444  bj(i) = ( i + alpha * mod ( i, 2 ) ) / 2.0;
445  end
446  bj(1:m) = sqrt ( bj(1:m) );
447 
448  elseif ( kind == 7 )
449 
450  ab = alpha;
451  zemu = 2.0 / ( ab + 1.0 );
452 
453  aj(1:m) = 0.0;
454 
455  for i = 1 : m
456  abi = i + ab * mod(i,2);
457  abj = 2 * i + ab;
458  bj(i) = abi * abi / ( abj * abj - 1.0 );
459  end
460  bj(1:m) = sqrt ( bj(1:m) );
461 
462  elseif ( kind == 8 )
463 
464  ab = alpha + beta;
465  zemu = gamma ( alpha + 1.0 ) * gamma ( - ( ab + 1.0 ) ) ...
466  / gamma ( - beta );
467  apone = alpha + 1.0;
468  aba = ab * apone;
469  aj(1) = - apone / ( ab + 2.0 );
470  bj(1) = - aj(1) * ( beta + 1.0 ) / ( ab + 2.0 ) / ( ab + 3.0 );
471  for i = 2 : m
472  abti = ab + 2.0 * i;
473  aj(i) = aba + 2.0 * ( ab + i ) * ( i - 1 );
474  aj(i) = - aj(i) / abti / ( abti - 2.0 );
475  end
476 
477  for i = 2 : m - 1
478  abti = ab + 2.0 * i;
479  bj(i) = i * ( alpha + i ) / ( abti - 1.0 ) * ( beta + i ) ...
480  / ( abti^2 ) * ( ab + i ) / ( abti + 1.0 );
481  end
482 
483  bj(m) = 0.0;
484  bj(1:m) = sqrt ( bj(1:m) );
485 
486  end
487 
488  return
489 end

492 %*****************************************************************************80
493 %
494 %% IMTQLX diagonalizes a symmetric tridiagonal matrix.
495 %
496 % Discussion:
497 %
498 % This routine is a slightly modified version of the EISPACK routine to
499 % perform the implicit QL algorithm on a symmetric tridiagonal matrix.
500 %
501 % The authors thank the authors of EISPACK for permission to use this
502 % routine.
503 %
504 % It has been modified to produce the product Q' * Z, where Z is an input
505 % vector and Q is the orthogonal matrix diagonalizing the input matrix.
506 % The changes consist (essentialy) of applying the orthogonal transformations
507 % directly to Z as they are generated.
508 %
509 % Licensing:
510 %
511 % This code is distributed under the GNU LGPL license.
512 %
513 % Modified:
514 %
515 % 04 January 2010
516 %
517 % Author:
518 %
519 % Original FORTRAN77 version by Sylvan Elhay, Jaroslav Kautsky.
520 % MATLAB version by John Burkardt.
521 %
522 % Reference:
523 %
524 % Sylvan Elhay, Jaroslav Kautsky,
525 % Algorithm 655: IQPACK, FORTRAN Subroutines for the Weights of
526 % Interpolatory Quadrature,
527 % ACM Transactions on Mathematical Software,
528 % Volume 13, Number 4, December 1987, pages 399-415.
529 %
530 % Roger Martin, James Wilkinson,
531 % The Implicit QL Algorithm,
532 % Numerische Mathematik,
533 % Volume 12, Number 5, December 1968, pages 377-383.
534 %
535 % Parameters:
536 %
537 % Input, integer N, the order of the matrix.
538 %
539 % Input, real D(N), the diagonal entries of the matrix.
540 %
541 % Input, real E(N), the subdiagonal entries of the
542 % matrix, in entries E(1) through E(N-1).
543 %
544 % Input, real Z(N), a vector to be operated on.
545 %
546 % Output, real D(N), the diagonal entries of the diagonalized matrix.
547 %
548 % Output, real Z(N), the value of Q' * Z, where Q is the matrix that
549 % diagonalizes the input symmetric tridiagonal matrix.
550 %
551  itn = 30;
552 
553  prec = eps;
554 
555  if ( n == 1 )
556  return
557  end
558 
559  e(n) = 0.0;
560 
561  for l = 1 : n
562 
563  j = 0;
564 
565  while ( 1 )
566 
567  for m = l : n
568 
569  if ( m == n )
570  break
571  end
572 
573  if ( abs ( e(m) ) <= prec * ( abs ( d(m) ) + abs ( d(m+1) ) ) )
574  break
575  end
576 
577  end
578 
579  p = d(l);
580 
581  if ( m == l )
582  break
583  end
584 
585  if ( j == itn )
586  fprintf ( 1, '\n' );
587  fprintf ( 1, 'IMTQLX - Fatal error!\n' );
588  fprintf ( 1, ' Iteration limit exceeded.\n' );
589  error ( 'IMTQLX - Fatal error!' );
590  end
591 
592  j = j + 1;
593  g = ( d(l+1) - p ) / ( 2.0 * e(l) );
594  r = sqrt ( g * g + 1.0 );
595  g = d(m) - p + e(l) / ( g + r8_sign ( g ) * abs ( r ) );
596  s = 1.0;
597  c = 1.0;
598  p = 0.0;
599  mml = m - l;
600 
601  for ii = 1 : mml
602 
603  i = m - ii;
604  f = s * e(i);
605  b = c * e(i);
606 
607  if ( abs ( f ) >= abs ( g ) )
608  c = g / f;
609  r = sqrt ( c * c + 1.0 );
610  e(i+1) = f * r;
611  s = 1.0 / r;
612  c = c * s;
613  else
614  s = f / g;
615  r = sqrt ( s * s + 1.0 );
616  e(i+1) = g * r;
617  c = 1.0 / r;
618  s = s * c;
619  end
620 
621  g = d(i+1) - p;
622  r = ( d(i) - g ) * s + 2.0 * c * b;
623  p = s * r;
624  d(i+1) = g + p;
625  g = c * r - b;
626  f = z(i+1);
627  z(i+1) = s * z(i) + c * f;
628  z(i) = c * z(i) - s * f;
629 
630  end
631 
632  d(l) = d(l) - p;
633  e(l) = g;
634  e(m) = 0.0;
635 
636  end
637 
638  end
639 
640  for ii = 2 : n
641 
642  i = ii - 1;
643  k = i;
644  p = d(i);
645 
646  for j = ii : n
647  if ( d(j) < p )
648  k = j;
649  p = d(j);
650  end
651  end
652 
653  if ( k ~= i )
654  d(k) = d(i);
655  d(i) = p;
656  p = z(i);
657  z(i) = z(k);
658  z(k) = p;
659  end
660 
661  end
662 
663  return
664 end
