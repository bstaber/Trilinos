3 %*****************************************************************************80
4 %
5 %% LF_FUNCTION_VALUES returns some values of the Laguerre function Lf(n,alpha,x).
6 %
7 % Discussion:
8 %
9 % In Mathematica, the function can be evaluated by:
10 %
11 % LaguerreL[n,a,x]
12 %
13 % The functions satisfy the following differential equation:
14 %
15 % X * Y'' + (ALPHA+1-X) * Y' + N * Y = 0
16 %
17 % Function values can be generated by the recursion:
18 %
19 % Lf(0,ALPHA,X) = 1
20 % Lf(1,ALPHA,X) = 1+ALPHA-X
21 %
22 % Lf(N,ALPHA,X) = ( (2*N-1+ALPHA-X) * Lf(N-1,ALPHA,X)
23 % - (N-1+ALPHA) * Lf(N-2,ALPHA,X) ) / N
24 %
25 % The parameter ALPHA is required to be greater than -1.
26 %
27 % For ALPHA = 0, the generalized Laguerre function Lf(N,ALPHA,X)
28 % is equal to the Laguerre polynomial L(N,X).
29 %
30 % For ALPHA integral, the generalized Laguerre function
31 % Lf(N,ALPHA,X) equals the associated Laguerre polynomial Lm(N,ALPHA,X).
32 %
